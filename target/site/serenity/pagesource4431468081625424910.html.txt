    function scheduleNavigationLog() {
        var navigationLog = JSON.parse(localStorage.getItem("navigation_log"));//NOSONAR
        if (!navigationLog || navigationLog.length == 0) {
            navigationLog = [];
        }
        var currentTime = (new Date()).toISOString().slice(0, 19).replace('T', ' ');
        var latestLog = sessionStorage.getItem("navigation_log_latest");
        if (!latestLog) {
            latestLog = localStorage.getItem("navigation_log_latest");//NOSONAR
        }
        latestLog = JSON.parse(latestLog);
        navigationLog.push({from: latestLog.to, to: '/auth/login', time: currentTime});

        var date = new Date();
        date.setTime(date.getTime() + (24*60*60*1000));
        expires = "; expires=" + date.toUTCString();
        document.cookie = "navigation_log" + "=" + JSON.stringify(navigationLog) + expires + "; path=/";
        localStorage.setItem("navigation_log", JSON.stringify([]));//NOSONAR
        localStorage.setItem("navigation_logged_time", new Date());//NOSONAR
    }
